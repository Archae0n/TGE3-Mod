# Check if current scope country has comms with any other regular empire
# THIS = COUNTRY
kiyo_spy_any_regular_comms = {
	is_country_type = default
	any_country = {
		is_country_type = default
		NOT = { is_country = prev }
		prev = { has_communications = prev }
	}
}

			has_technology = "tech_colossus"

		}
	}
}

# Checks if ship is in valid location for spying
# PREVPREV = COUNTRY (EMPIRE TO SPY ON)
# PREV = COUNTRY (OWNER OF SHIP)
# THIS = SHIP
kiyo_spy_valid_fleet = {
	if = {
		limit = {
			exists = fleet
			exists = leader
		}
		fleet = {
			if = { limit = { exists = orbit }
				orbit = { is_primary_star = yes }
			}
			else = { always = no }
			kiyo_spy_fleet_order_valid = yes
			prev = {
				OR = {
					AND = {
						is_ship_class = shipclass_transport
						fleet = { num_ships >= 20 }
						leader = {
							leader_class = general
							is_event_leader = no
							#has_leader_flag = kiyo_spy_trigger_spy
						}
					}
					AND = {
						is_ship_class = shipclass_science_ship
						leader = {
							leader_class = scientist
							is_event_leader = no
							#has_leader_flag = kiyo_spy_trigger_spy
							has_skill >= 4
						}
					}
				}
			}
		}
	}
	else = { always = no }
	prevprev = {
		any_system_within_border = {
			prevprev = {
				distance = { source = prev type = hyperlane use_bypasses = no min_jumps = 0 max_jumps = 3 }
			}
		}
	}
}

# Checks for valid fleet at location that has stayed for a long while
# ROOT = COUNTRY
kiyo_spy_valid_fleet_wants_to_spy = {
	root = {
		any_owned_ship = {
			kiyo_spy_valid_fleet = yes
			if = { limit = { exists = fleet }
				fleet = {
					OR = {
						AND = {
							owner = { is_ai = no }
							has_fleet_flag = kiyo_spy_fleet_wants_to_spy@prevprevprev
							NOT = { has_fleet_flag = kiyo_spy_fleet_maybe_spying }
						}
						AND = {
							owner = { is_ai = yes }
							has_fleet_flag = kiyo_spy_ai_spying
						}
					}
				}
			}
		}
	}
}
